<link rel="stylesheet" href="путь_до_модуля/css/umcconsultant.css" type="text/css" />
  <script src="путь_до_модуля/js/jquery-1.11.3.min.js" type="text/javascript"></script>
  <script src="путь_до_модуля/js/jquery.cookie.js" type="text/javascript"></script>
  <script src="путь_до_модуля/js/umcconsultant.js" type="text/javascript"></script>
